function rgb(r, g, b) {
  const rInTheRange = getNumberInTheRange(r);
  const gInTheRange = getNumberInTheRange(g);
  const bInTheRange = getNumberInTheRange(b);

  return changeToHex(rInTheRange) + changeToHex(gInTheRange) + changeToHex(bInTheRange);
}

function getNumberInTheRange(number) {
  if (number < 0) {
    return 0;
  }

  if (number > 255) {
    return 255;
  }

  return number;
}

function changeToHex(number) {
  const hexNumber = number.toString(16).toUpperCase();

  if (hexNumber.length < 2) {
    return '0' + hexNumber;
  }

  return hexNumber;
}
